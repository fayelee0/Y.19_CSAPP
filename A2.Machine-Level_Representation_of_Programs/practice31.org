#+BEGIN_SRC c

void switcher(long a, long b, long c, long* dest)
{
    long val;
    switch (a) {
    case 5:  /* case A */
        c = b ^ 15;
        /* fall through */
    case 0:  /* case B */
        val = 112 + c;
        break;
    case 2:  /* case C */
    case 7:  /* case D */
        val = (c + b) << 2;
        break;
    case 4:  /* case E */
        val = a;
        break;
    default:
        val = b;
    }
    *dest = val;
}
